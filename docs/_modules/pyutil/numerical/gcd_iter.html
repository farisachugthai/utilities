    """Find the greatest common denominator with 2 arbitrary integers.

    Parameters
    ----------
    a : int
        User provided integer
    b : int
        User provided integer

    Returns
    -------
    gcd : int
        Greatest common denominator.

    """
    orig_b = b
    orig_a = a
    if a > b:
        while b > 0:
            if a % b == 0 and orig_b % b == 0:
                return b
            b -= 1
    else:
        while a > 0:
            if b % a == 0 and orig_a % a == 0:
                return a
            a -= 1
